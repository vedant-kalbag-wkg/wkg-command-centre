-- Query:
SELECT
    pg_database_size(datname) AS db_size,
    datid
FROM
    pg_stat_database
WHERE datname IN (
    SELECT datname
    FROM pg_database
    WHERE datconnlimit != -2
    AND datname <> 'postgres'
    AND NOT datistemplate
    ORDER BY oid
    LIMIT 10
)
