# Phase 3 Implementation Plan — Caching layer + RSC streaming

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.
>
> Each task uses superpowers:test-driven-development. Each task ends with a commit. Stage gates are BLOCKING — do not proceed past them if the gate fails.

**Goal:** Cache analytics query results per-day and stream RSC content via Suspense across all 9 heavy `/analytics/*` pages, collapsing warm-load p95 from 1318 ms (Phase 2 baseline) to ≤ 300 ms.

**Architecture:** Wrap each analytics query function with Next.js `unstable_cache` (TTL=86400s, tags=`['analytics', 'analytics:<page>']`, scope key collapses internal users to a single shared entry). Each `page.tsx` decomposes into Suspense-bounded data islands with per-island `<ErrorBoundary>` and pixel-faithful skeletons. Admin escape hatch at `/admin/cache` with audit-logged tag-purge.

**Tech Stack:** Next.js 15 App Router (`unstable_cache`, `revalidateTag`, `<Suspense>`), React 19, TypeScript, vitest (unit + Neon-dev integration), Playwright (remote preview against Vercel), drizzle-orm 0.45, postgres-js, shadcn `<Skeleton>`.

```ts
export interface AnalyticsFilters {
  dateFrom?: Date | null;
  dateTo?: Date | null;
  hotelGroups?: string[];
  locationGroups?: string[];
  regions?: string[];
}

export interface CanonicalFilters {
  dateFrom: string | null;
  dateTo: string | null;
  hotelGroups: string[];
  locationGroups: string[];
  regions: string[];
}

function dateOnly(d: Date | null | undefined): string | null {
  if (!d) return null;
  return d.toISOString().slice(0, 10);
}

function sortedOrEmpty(xs: string[] | undefined): string[] {
  return [...(xs ?? [])].sort();
}

export function canonicaliseFilters(f: AnalyticsFilters): CanonicalFilters {
  return {
    dateFrom: dateOnly(f.dateFrom ?? null),
    dateTo:   dateOnly(f.dateTo   ?? null),
    hotelGroups:    sortedOrEmpty(f.hotelGroups),
    locationGroups: sortedOrEmpty(f.locationGroups),
    regions:        sortedOrEmpty(f.regions),
  };
}
```

```ts
import { createHash } from 'node:crypto';
import { requireRole } from '@/lib/auth';
import { getActiveLocationIds } from '@/lib/analytics/active-locations';

export async function getCacheScopeKey(): Promise<string> {
  const session = await requireRole('admin', 'member', 'external');
  if (session.user.kind === 'external') {
    const ids = await getActiveLocationIds();
    const hash = createHash('sha1').update([...ids].sort().join(',')).digest('hex').slice(0, 16);
    return `ext:${hash}`;
  }
  return '__internal__';
}
```

```ts
type AsyncFn<TArgs extends unknown[], TR> = (...args: TArgs) => Promise<TR>;

export function withStats<TArgs extends unknown[], TR>(
  name: string,
  fn: AsyncFn<TArgs, TR>,
): AsyncFn<TArgs, TR> {
  return async (...args: TArgs) => {
    const t0 = performance.now();
    const result = await fn(...args);
    const dt = performance.now() - t0;
    if (process.env.LOG_CACHE_STATS) {
      console.log(JSON.stringify({ q: name, dt: Math.round(dt * 100) / 100, hit: dt < 10 }));
    }
    return result;
  };
}
```

```ts
'use server';
import { revalidateTag } from 'next/cache';
import { requireRole } from '@/lib/auth';
import { writeAuditLog } from '@/lib/audit';

export type AnalyticsPageTag =
  | 'portfolio' | 'hotel-groups' | 'regions' | 'heat-map'
  | 'location-groups' | 'maturity' | 'trend-builder' | 'pivot-table' | 'compare';

export async function purgeAnalyticsCache(scope: 'all' | AnalyticsPageTag = 'all') {
  const session = await requireRole('admin');
  const tag = scope === 'all' ? 'analytics' : `analytics:${scope}`;
  revalidateTag(tag);
  await writeAuditLog({
    actorId: session.user.id,
    actorName: session.user.name,
    entityType: 'cache',
    entityId: tag,
    action: 'purge',
  });
  return { success: true as const, tag };
}
```
