**Step 1 — Create `src/lib/stores/view-engine-store.ts`:**
Use Zustand v5 `create` (import from "zustand"). Create a **factory function** `createViewEngineStore(entityType: string)` that returns a separate store instance per entity type. This prevents filter state bleeding between /kiosks and /locations (Pitfall 3 from RESEARCH.md).

Store shape:
```typescript
import { create } from "zustand";
import type { ColumnFiltersState, SortingState, VisibilityState } from "@tanstack/react-table";

interface ViewEngineState {
  entityType: string;
  columnFilters: ColumnFiltersState;
  sorting: SortingState;
  grouping: string[];
  columnVisibility: VisibilityState;
  globalFilter: string;
  rowSelection: Record<string, boolean>;
  // Actions
  setColumnFilters: (filters: ColumnFiltersState) => void;
  setSorting: (sorting: SortingState) => void;
  setGrouping: (grouping: string[]) => void;
  setColumnVisibility: (visibility: VisibilityState) => void;
  setGlobalFilter: (filter: string) => void;
  setRowSelection: (selection: Record<string, boolean>) => void;
  applyView: (config: { columnFilters?: ColumnFiltersState; sorting?: SortingState; grouping?: string[]; columnVisibility?: VisibilityState }) => void;
  resetToDefaults: () => void;
  getCurrentConfig: () => { columnFilters: ColumnFiltersState; sorting: SortingState; grouping: string[]; columnVisibility: VisibilityState };
}
```

Default sorting: `[{ id: "createdAt", desc: true }]` (newest first per UI-SPEC).
Default grouping: `[]` (no grouping).
Default columnVisibility: `{}` (all visible — column definitions control defaults).

Export two pre-created stores:
```typescript
export const useKioskViewStore = createViewEngineStore("kiosk");
export const useLocationViewStore = createViewEngineStore("location");
```

**Step 2 — Create `src/app/(app)/kiosks/views-actions.ts` and `src/app/(app)/locations/views-actions.ts`:**
Server actions for saved views CRUD. Both files follow same pattern:

```typescript
"use server";
import { z } from "zod/v4";
import { db } from "@/db";
import { userViews } from "@/db/schema";
import { getSessionOrThrow } from "@/lib/rbac";
import { eq, and } from "drizzle-orm";

export async function saveView(name: string, config: ViewConfig) {
  const session = await getSessionOrThrow();
  const [view] = await db.insert(userViews).values({
    userId: session.user.id,
    name,
    entityType: "kiosk", // or "location"
    config,
  }).returning();
  return { success: true, id: view.id };
}

export async function listSavedViews() {
  const session = await getSessionOrThrow();
  return db.select().from(userViews)
    .where(and(eq(userViews.userId, session.user.id), eq(userViews.entityType, "kiosk")));
}

export async function updateView(viewId: string, name: string, config: ViewConfig) { ... }
export async function deleteView(viewId: string) { ... }
```
