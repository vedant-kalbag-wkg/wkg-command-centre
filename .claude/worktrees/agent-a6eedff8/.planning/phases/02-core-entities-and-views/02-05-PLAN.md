**Step 1 — Create `src/app/(app)/kiosks/bulk-actions.ts`:**
```typescript
"use server";
import { z } from "zod/v4";
import { db } from "@/db";
import { kiosks } from "@/db/schema";
import { requireRole } from "@/lib/rbac";
import { writeAuditLog } from "@/lib/audit";
import { inArray, eq } from "drizzle-orm";

export async function bulkUpdateKiosks(
  kioskIds: string[],
  updates: { field: string; value: unknown }
) {
  const session = await requireRole("admin", "member");
  // Validate field is an allowed bulk-edit field
  const allowedFields = ["pipelineStageId", "regionGroup", "cmsConfigStatus", "deploymentPhaseTags"];
  if (!allowedFields.includes(updates.field)) {
    return { error: "Field not allowed for bulk edit" };
  }
  // Fetch current values for audit log
  const currentKiosks = await db.select({ id: kiosks.id, [updates.field]: kiosks[updates.field as keyof typeof kiosks] })
    .from(kiosks).where(inArray(kiosks.id, kioskIds));
  // Update all
  await db.update(kiosks)
    .set({ [updates.field]: updates.value, updatedAt: new Date() })
    .where(inArray(kiosks.id, kioskIds));
  // Write audit for each
  for (const kiosk of currentKiosks) {
    await writeAuditLog({
      actorId: session.user.id,
      actorName: session.user.name,
      entityType: "kiosk",
      entityId: kiosk.id,
      entityName: kiosk.id,
      action: "update",
      field: updates.field,
      oldValue: String(kiosk[updates.field as keyof typeof kiosk] ?? ""),
      newValue: String(updates.value ?? ""),
    });
  }
  return { success: true, count: kioskIds.length };
}

export async function bulkArchiveKiosks(kioskIds: string[]) {
  const session = await requireRole("admin", "member");
  await db.update(kiosks)
    .set({ archivedAt: new Date(), updatedAt: new Date() })
    .where(inArray(kiosks.id, kioskIds));
  for (const id of kioskIds) {
    await writeAuditLog({ actorId: session.user.id, actorName: session.user.name,
      entityType: "kiosk", entityId: id, entityName: id, action: "archive" });
  }
  return { success: true, count: kioskIds.length };
}
```

**Step 3 — Create `src/components/table/csv-export.ts`:**
Utility function (not a component):
```typescript
import Papa from "papaparse";
import type { Table } from "@tanstack/react-table";

export function exportTableToCSV<T>(table: Table<T>, fileName: string) {
  const rows = table.getFilteredRowModel().rows;
  const visibleColumns = table.getVisibleFlatColumns()
    .filter(col => col.id !== "select" && col.id !== "actions");
  const headers = visibleColumns.map(col => String(col.columnDef.header ?? col.id));
  const data = rows
    .filter(row => !row.getIsGrouped())
    .map(row => visibleColumns.map(col => {
      const val = row.getValue(col.id);
      return val instanceof Date ? val.toLocaleDateString() : String(val ?? "");
    }));
  const csv = Papa.unparse({ fields: headers, data });
  const blob = new Blob(["\uFEFF" + csv], { type: "text/csv;charset=utf-8;" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = `${fileName}-${new Date().toISOString().split("T")[0]}.csv`;
  a.click();
  URL.revokeObjectURL(url);
}
```
Note: explicit `fields` array for deterministic column order (avoids Pitfall 7 from RESEARCH.md). UTF-8 BOM `\uFEFF` for Excel compatibility.

Server actions for audit log queries:

1. `fetchAuditEntries(params: { entityType?: string, entityId?: string, actorId?: string, dateFrom?: string, dateTo?: string, cursor?: string, limit?: number })` — Returns audit entries matching filters, ordered by createdAt DESC, limit 20 (default). Cursor-based pagination: `cursor` is the `id` of the last entry from previous page. Return `{ entries: AuditEntry[], hasMore: boolean, remainingCount: number }`. No requireRole check for entity-specific queries (any logged-in user). For global queries (no entityId), `requireRole("admin")`.

2. `fetchAuditActors()` — Returns distinct actor names from auditLogs for the filter dropdown. `requireRole("admin")`.
