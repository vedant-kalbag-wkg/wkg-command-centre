```typescript
export const products = pgTable("products", {
  id: uuid("id").primaryKey().defaultRandom(),
  name: text("name").notNull().unique(),
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow().notNull(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow().notNull(),
});
```

**Table 2: `providers`** (per D-12)
```typescript
export const providers = pgTable("providers", {
  id: uuid("id").primaryKey().defaultRandom(),
  name: text("name").notNull().unique(),
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow().notNull(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow().notNull(),
});
```

**Table 3: `location_products`** (per D-12, D-13)
```typescript
export const locationProducts = pgTable("location_products", {
  id: uuid("id").primaryKey().defaultRandom(),
  locationId: uuid("location_id").notNull().references(() => locations.id),
  productId: uuid("product_id").notNull().references(() => products.id),
  providerId: uuid("provider_id").references(() => providers.id),
  availability: text("availability").notNull().default("unavailable"),
  commissionTiers: jsonb("commission_tiers").$type<Array<{ minRevenue: number; maxRevenue: number | null; rate: number }>>(),
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow().notNull(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).defaultNow().notNull(),
});
```

After editing the schema file, run `npx drizzle-kit push` to push the new tables to the database.

Key constraints:
- `products.name` is UNIQUE (per D-12)
- `providers.name` is UNIQUE (per D-12)
- `locationProducts.providerId` is nullable (per D-12 — not all products have a provider assigned)
- `locationProducts.availability` defaults to "unavailable" (per D-15 — new products start as unavailable)
- `commissionTiers` JSONB stores array of `{ minRevenue, maxRevenue (nullable), rate }` (per D-13)
