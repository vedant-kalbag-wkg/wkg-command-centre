**B. Create `src/app/(app)/kiosks/view-tabs-client.tsx`** ("use client"):

This client component handles tab switching with URL synchronization.

1. Use `useRouter()` and `usePathname()` from next/navigation
2. When tab changes: `router.push(\`${pathname}?view=${newTab}\`, { scroll: false })`
3. Render 4 tabs: Table, Kanban, Gantt, Calendar
4. Use `Tabs` component with `value={activeView}` (controlled) and `onValueChange` handler that pushes URL

```typescript
"use client";
import { useRouter, usePathname } from "next/navigation";
import { Tabs, TabsList, TabsTrigger, TabsContent } from "@/components/ui/tabs";
import { KioskTable } from "@/components/kiosks/kiosk-table";
import { KioskKanban } from "@/components/kiosks/kiosk-kanban";
import { GanttTab } from "./gantt-tab";
import { CalendarTab } from "./calendar-tab";

interface ViewTabsClientProps {
  activeView: string;
  kiosks: any[]; // KioskWithRelations[]
  stages: any[];
  installations: any[]; // InstallationWithRelations[]
}

export function ViewTabsClient({ activeView, kiosks, stages, installations }: ViewTabsClientProps) {
  const router = useRouter();
  const pathname = usePathname();

  function handleTabChange(value: string) {
    router.push(`${pathname}?view=${value}`, { scroll: false });
  }

  return (
    <Tabs value={activeView} onValueChange={handleTabChange}>
      <TabsList variant="line" className="mb-4">
        <TabsTrigger value="table">Table</TabsTrigger>
        <TabsTrigger value="kanban">Kanban</TabsTrigger>
        <TabsTrigger value="gantt">Gantt</TabsTrigger>
        <TabsTrigger value="calendar">Calendar</TabsTrigger>
      </TabsList>
      <TabsContent value="table">
        <KioskTable data={kiosks} />
      </TabsContent>
      <TabsContent value="kanban">
        <KioskKanban kiosks={kiosks} stages={stages} />
      </TabsContent>
      <TabsContent value="gantt">
        <GanttTab installations={installations} />
      </TabsContent>
      <TabsContent value="calendar">
        <CalendarTab installations={installations} kiosks={kiosks} />
      </TabsContent>
    </Tabs>
  );
}
```

5. IMPORTANT: Use `value` (controlled mode) not `defaultValue` — so the tab reacts to URL changes on navigation.

6. Ensure independent filters: Each tab component manages its own filter state via its own Zustand store (useKioskViewStore for table, useGanttStore for gantt, useCalendarStore for calendar). No cross-tab state sync needed — each store is independent.
